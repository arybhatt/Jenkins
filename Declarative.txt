pipeline {
    agent any

    environment {
        IMAGE_NAME = "my-hello-world-app"
        CONTAINER_NAME = "web-container"
        PORT = "3000"
        // Pointing specifically to your filename in the repo
        DOCKERFILE_PATH = "Dockerfile.txt" 
    }

    stages {
        // Removed the manual 'Checkout' stage as Jenkins does this automatically

        stage('Cleanup Old Container') {
            steps {
                script {
                    echo 'Cleaning up existing containers...'
                    // || true ensures the pipeline doesn't fail if the container doesn't exist yet
                    sh "docker rm -f ${CONTAINER_NAME} || true"
                }
            }
        }

        stage('Build Docker Image') {
            steps {
                echo 'Building the Docker image...'
                // Using -f to specify your Dockerfile.txt
                sh "docker build -f ${DOCKERFILE_PATH} -t ${IMAGE_NAME} ."
            }
        }

        stage('Run Container') {
            steps {
                echo "Deploying to port ${PORT}..."
                // -d runs it in the background
                sh "docker run -d --name ${CONTAINER_NAME} -p ${PORT}:3000 ${IMAGE_NAME}"
            }
        }
    }

    post {
        success {
            echo "Successfully deployed! Your app is now live."
            echo "Visit http://your-server-ip:${PORT}"
        }
        failure {
            echo "The build was dead in the water. Check the console output above for errors."
        }
        always {
            echo "Pipeline execution finished."
        }
    }
}
